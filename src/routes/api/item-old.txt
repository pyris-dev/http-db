import { config, dbManager } from "../../index.js";
import { checkAuth } from "../../util.js";
import { RouteHandler, RouteType } from "../index.js";

/** 
 * Item API Route Handler
 * This handler manages GET and POST requests for database items.
*/
class ItemApiRouteHandler extends RouteHandler {
    static type = RouteType.METHOD;

    static onGet(req: Request): Response {
        const params = (req as any).params || {};
        const table = params.table as string;
        const id = params.id as string;
        const item = dbManager.get(table, id);
        if (!item) return new Response('Row not found', { status: 404 });
        return Response.json({ item });
    }

    static async onPut(req: Request): Promise<Response> {
        if (!checkAuth(req, config)) return new Response('Unauthorized', { status: 401 });

        const params = (req as any).params || {};
        const table = params.table as string;
        const id = params.id as string;
        try {
            const patch = (await req.json()) as Record<string, unknown>;
            const updated = dbManager.update(table, id, patch);
            if (!updated) return new Response('Row not found', { status: 404 });
            return Response.json({ updated });
        } catch {
            return Response.json({ error: 'Invalid JSON' }, 400);
        }
    }

    static onDelete(req: Request): Response {
        if (!checkAuth(req, config)) return new Response('Unauthorized', { status: 401 });

        const params = (req as any).params || {};
        const table = params.table as string;
        const id = params.id as string;
        const ok = dbManager.remove(table, id);
        if (!ok) return new Response('Row not found', { status: 404 });
        return new Response(null, { status: 204 });
    }
}

export { ItemApiRouteHandler };
